71/* Convert OpenSSL RSA private key to android pre-computed RSAPublicKey format */
72static int RSA_to_RSAPublicKey(RSA *rsa, RSAPublicKey *pkey)
73{
74 int ret = 1;
75 unsigned int i;
76
77 BN_CTX* ctx = BN_CTX_new();
78 BIGNUM* r32 = BN_new();
79 BIGNUM* rr = BN_new();
80 BIGNUM* r = BN_new();
81 BIGNUM* rem = BN_new();
82 BIGNUM* n = BN_new();
83 BIGNUM* n0inv = BN_new();
84
85 if (RSA_size(rsa) != RSANUMBYTES) {
86 ret = 0;
87 goto out;
88 }
89
90 BN_set_bit(r32, 32);
91 BN_copy(n, rsa->n);
92 BN_set_bit(r, RSANUMWORDS * 32);
93 BN_mod_sqr(rr, r, n, ctx);
94 BN_div(NULL, rem, n, r32, ctx);
95 BN_mod_inverse(n0inv, rem, r32, ctx);
96
97 pkey->len = RSANUMWORDS;
98 pkey->n0inv = 0 - BN_get_word(n0inv);
99 for (i = 0; i < RSANUMWORDS; i++) {
100 BN_div(rr, rem, rr, r32, ctx);
101 pkey->rr[i] = BN_get_word(rem);
102 BN_div(n, rem, n, r32, ctx);
103 pkey->n[i] = BN_get_word(rem);
104 }
105 pkey->exponent = BN_get_word(rsa->e);
106
107out:
108 BN_free(n0inv);
109 BN_free(n);
110 BN_free(rem);
111 BN_free(r);
112 BN_free(rr);
113 BN_free(r32);
114 BN_CTX_free(ctx);
115
116 return ret;
117}

119static void get_user_info(char *buf, size_t len)
120{
121 char hostname[1024], username[1024];
122 int ret = -1;
123
124 if (getenv("HOSTNAME") != NULL) {
125 strncpy(hostname, getenv("HOSTNAME"), sizeof(hostname));
126 hostname[sizeof(hostname)-1] = '\0';
127 ret = 0;
128 }
129
130#ifndef _WIN32
131 if (ret < 0)
132 ret = gethostname(hostname, sizeof(hostname));
133#endif
134 if (ret < 0)
135 strcpy(hostname, "unknown");
136
137 ret = -1;
138
139 if (getenv("LOGNAME") != NULL) {
140 strncpy(username, getenv("LOGNAME"), sizeof(username));
141 username[sizeof(username)-1] = '\0';
142 ret = 0;
143 }
144
145#if !defined _WIN32 && !defined ADB_HOST_ON_TARGET
146 if (ret < 0)
147 ret = getlogin_r(username, sizeof(username));
148#endif
149 if (ret < 0)
150 strcpy(username, "unknown");
151
152 ret = snprintf(buf, len, " %s@%s", username, hostname);
153 if (ret >= (signed)len)
154 buf[len - 1] = '\0';
155}

157static int write_public_keyfile(RSA *private_key, const char *private_key_path)
158{
159 RSAPublicKey pkey;
160 FILE *outfile = NULL;
161 char path[PATH_MAX], info[MAX_PAYLOAD_V1];
162 uint8_t* encoded = nullptr;
163 size_t encoded_length;
164 int ret = 0;
165
166 if (snprintf(path, sizeof(path), "%s.pub", private_key_path) >=
167 (int)sizeof(path)) {
168 D("Path too long while writing public key");
169 return 0;
170 }
171
172 if (!RSA_to_RSAPublicKey(private_key, &pkey)) {
173 D("Failed to convert to publickey");
174 return 0;
175 }
176
177 outfile = fopen(path, "w");
178 if (!outfile) {
179 D("Failed to open '%s'", path);
180 return 0;
181 }
182
183 D("Writing public key to '%s'", path);
184
185#if defined(OPENSSL_IS_BORINGSSL)
186 if (!EVP_EncodedLength(&encoded_length, sizeof(pkey))) {
187 D("Public key too large to base64 encode");
188 goto out;
189 }
190#else
191 /* While we switch from OpenSSL to BoringSSL we have to implement
192 * |EVP_EncodedLength| here. */
193 encoded_length = 1 + ((sizeof(pkey) + 2) / 3 * 4);
194#endif
195
196 encoded = new uint8_t[encoded_length];
197 if (encoded == nullptr) {
198 D("Allocation failure");
199 goto out;
200 }
201
202 encoded_length = EVP_EncodeBlock(encoded, (uint8_t*) &pkey, sizeof(pkey));
203 get_user_info(info, sizeof(info));
204
205 if (fwrite(encoded, encoded_length, 1, outfile) != 1 ||
206 fwrite(info, strlen(info), 1, outfile) != 1) {
207 D("Write error while writing public key");
208 goto out;
209 }
210
211 ret = 1;
212
213 out:
214 if (outfile != NULL) {
215 fclose(outfile);
216 }
217 delete[] encoded;
218 return ret;
219}

221static int generate_key(const char *file)
222{
223 EVP_PKEY* pkey = EVP_PKEY_new();
224 BIGNUM* exponent = BN_new();
225 RSA* rsa = RSA_new();
226 mode_t old_mask;
227 FILE *f = NULL;
228 int ret = 0;
229
230 D("generate_key '%s'", file);
231
232 if (!pkey || !exponent || !rsa) {
233 D("Failed to allocate key");
234 goto out;
235 }
236
237 BN_set_word(exponent, RSA_F4);
238 RSA_generate_key_ex(rsa, 2048, exponent, NULL);
239 EVP_PKEY_set1_RSA(pkey, rsa);
240
241 old_mask = umask(077);
242
243 f = fopen(file, "w");
244 if (!f) {
245 D("Failed to open '%s'", file);
246 umask(old_mask);
247 goto out;
248 }
249
250 umask(old_mask);
251
252 if (!PEM_write_PrivateKey(f, pkey, NULL, NULL, 0, NULL, NULL)) {
253 D("Failed to write key");
254 goto out;
255 }
256
257 if (!write_public_keyfile(rsa, file)) {
258 D("Failed to write public key");
259 goto out;
260 }
261
262 ret = 1;
263
264out:
265 if (f)
266 fclose(f);
267 EVP_PKEY_free(pkey);
268 RSA_free(rsa);
269 BN_free(exponent);
270 return ret;
271}

273static int read_key(const char *file, struct listnode *list)
274{
275 D("read_key '%s'", file);
276
277 FILE* fp = fopen(file, "r");
278 if (!fp) {
279 D("Failed to open '%s': %s", file, strerror(errno));
280 return 0;
281 }
282
283 adb_private_key* key = new adb_private_key;
284 key->rsa = RSA_new();
285
286 if (!PEM_read_RSAPrivateKey(fp, &key->rsa, NULL, NULL)) {
287 D("Failed to read key");
288 fclose(fp);
289 RSA_free(key->rsa);
290 delete key;
291 return 0;
292 }
293
294 fclose(fp);
295 list_add_tail(list, &key->node);
296 return 1;
297}

299static int get_user_keyfilepath(char *filename, size_t len)
300{
301 const char *format, *home;
302 char android_dir[PATH_MAX];
303 struct stat buf;
304#ifdef _WIN32
305 std::string home_str;
306 home = getenv("ANDROID_SDK_HOME");
307 if (!home) {
308 WCHAR path[MAX_PATH];
309 const HRESULT hr = SHGetFolderPathW(NULL, CSIDL_PROFILE, NULL, 0, path);
310 if (FAILED(hr)) {
311 D("SHGetFolderPathW failed: %s", android::base::SystemErrorCodeToString(hr).c_str());
312 return -1;
313 }
314 if (!android::base::WideToUTF8(path, &home_str)) {
315 return -1;
316 }
317 home = home_str.c_str();
318 }
319 format = "%s\\%s";
320#else
321 home = getenv("HOME");
322 if (!home)
323 return -1;
324 format = "%s/%s";
325#endif
326
327 D("home '%s'", home);
328
329 if (snprintf(android_dir, sizeof(android_dir), format, home,
330 ANDROID_PATH) >= (int)sizeof(android_dir))
331 return -1;
332
333 if (stat(android_dir, &buf)) {
334 if (adb_mkdir(android_dir, 0750) < 0) {
335 D("Cannot mkdir '%s'", android_dir);
336 return -1;
337 }
338 }
339
340 return snprintf(filename, len, format, android_dir, ADB_KEY_FILE);
341}

343static int get_user_key(struct listnode *list)
344{
345 struct stat buf;
346 char path[PATH_MAX];
347 int ret;
348
349 ret = get_user_keyfilepath(path, sizeof(path));
350 if (ret < 0 || ret >= (signed)sizeof(path)) {
351 D("Error getting user key filename");
352 return 0;
353 }
354
355 D("user key '%s'", path);
356
357 if (stat(path, &buf) == -1) {
358 if (!generate_key(path)) {
359 D("Failed to generate new key");
360 return 0;
361 }
362 }
363
364 return read_key(path, list);
365}

367static void get_vendor_keys(struct listnode* key_list) {
368 const char* adb_keys_path = getenv("ADB_VENDOR_KEYS");
369 if (adb_keys_path == nullptr) {
370 return;
371 }
372
373 for (const auto& path : android::base::Split(adb_keys_path, ENV_PATH_SEPARATOR_STR)) {
374 if (!read_key(path.c_str(), key_list)) {
375 D("Failed to read '%s'", path.c_str());
376 }
377 }
378}

380int adb_auth_sign(void *node, const unsigned char* token, size_t token_size,
381 unsigned char* sig)
382{
383 unsigned int len;
384 struct adb_private_key *key = node_to_item(node, struct adb_private_key, node);
385
386 if (token_size != TOKEN_SIZE) {
387 D("Unexpected token size %zd", token_size);
388 return 0;
389 }
390
391 if (!RSA_sign(NID_sha1, token, token_size, sig, &len, key->rsa)) {
392 return 0;
393 }
394
395 D("adb_auth_sign len=%d", len);
396 return (int)len;
397}

399void *adb_auth_nextkey(void *current)
400{
401 struct listnode *item;
402
403 if (list_empty(&key_list))
404 return NULL;
405
406 if (!current)
407 return list_head(&key_list);
408
409 list_for_each(item, &key_list) {
410 if (item == current) {
411 /* current is the last item, we tried all the keys */
412 if (item->next == &key_list)
413 return NULL;
414 return item->next;
415 }
416 }
417
418 return NULL;
419}

421int adb_auth_get_userkey(unsigned char *data, size_t len)
422{
423 char path[PATH_MAX];
424 int ret = get_user_keyfilepath(path, sizeof(path) - 4);
425 if (ret < 0 || ret >= (signed)(sizeof(path) - 4)) {
426 D("Error getting user key filename");
427 return 0;
428 }
429 strcat(path, ".pub");
430
431 // TODO(danalbert): ReadFileToString
432 // Note that on Windows, load_file() does not do CR/LF translation, but
433 // ReadFileToString() uses the C Runtime which uses CR/LF translation by
434 // default (by is overridable with _setmode()).
435 unsigned size;
436 char* file_data = reinterpret_cast<char*>(load_file(path, &size));
437 if (file_data == nullptr) {
438 D("Can't load '%s'", path);
439 return 0;
440 }
441
442 if (len < (size_t)(size + 1)) {
443 D("%s: Content too large ret=%d", path, size);
444 free(file_data);
445 return 0;
446 }
447
448 memcpy(data, file_data, size);
449 free(file_data);
450 file_data = nullptr;
451 data[size] = '\0';
452
453 return size + 1;
454}

460void adb_auth_init(void)
461{
462 int ret;
463
464 D("adb_auth_init");
465
466 list_init(&key_list);
467
468 ret = get_user_key(&key_list);
469 if (!ret) {
470 D("Failed to get user key");
471 return;
472 }
473
474 get_vendor_keys(&key_list);
475}
