17#include <limits.h>
18#include <stdio.h>
19#include <sys/types.h>
20#include <unistd.h>
21
22#include "adb.h"
23
24void get_my_path(char *exe, size_t maxLen)
25{
26 char proc[64];
27 snprintf(proc, sizeof proc, "/proc/%d/exe", getpid());
28 int err = readlink(proc, exe, maxLen - 1);
29 if(err > 0) {
30 exe[err] = '\0';
31 } else {
32 exe[0] = '\0';
33 }
34}
