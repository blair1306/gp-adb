34/* 7.18.1.1 Exact-width integer types */
35typedef signed char int8_t;
36typedef unsigned char uint8_t;
37typedef short int16_t;
38typedef unsigned short uint16_t;
39typedef int int32_t;
40typedef unsigned uint32_t;
41__MINGW_EXTENSION typedef long long int64_t;
42__MINGW_EXTENSION typedef unsigned long long uint64_t;
43
44/* 7.18.1.2 Minimum-width integer types */
45typedef signed char int_least8_t;
46typedef unsigned char uint_least8_t;
47typedef short int_least16_t;
48typedef unsigned short uint_least16_t;
49typedef int int_least32_t;
50typedef unsigned uint_least32_t;
51__MINGW_EXTENSION typedef long long int_least64_t;
52__MINGW_EXTENSION typedef unsigned long long uint_least64_t;
53
54/* 7.18.1.3 Fastest minimum-width integer types
55 * Not actually guaranteed to be fastest for all purposes
56 * Here we use the exact-width types for 8 and 16-bit ints.
57 */
58typedef signed char int_fast8_t;
59typedef unsigned char uint_fast8_t;
60typedef short int_fast16_t;
61typedef unsigned short uint_fast16_t;
62typedef int int_fast32_t;
63typedef unsigned int uint_fast32_t;
64__MINGW_EXTENSION typedef long long int_fast64_t;
65__MINGW_EXTENSION typedef unsigned long long uint_fast64_t;

67/* 7.18.1.5 Greatest-width integer types */
68__MINGW_EXTENSION typedef long long intmax_t;
69__MINGW_EXTENSION typedef unsigned long long uintmax_t;

174/* 7.18.4 Macros for integer constants */
175#if !defined(__cplusplus) || defined(__STDC_CONSTANT_MACROS) || \
176 defined(__GXX_EXPERIMENTAL_CXX0X__) || __cplusplus >= 201103L
177
178/* 7.18.4.1 Macros for minimum-width integer constants
179
180 Accoding to Douglas Gwyn <gwyn@arl.mil>:
181 "This spec was changed in ISO/IEC 9899:1999 TC1; in ISO/IEC
182 9899:1999 as initially published, the expansion was required
183 to be an integer constant of precisely matching type, which
184 is impossible to accomplish for the shorter types on most
185 platforms, because C99 provides no standard way to designate
186 an integer constant with width less than that of type int.
187 TC1 changed this to require just an integer constant
188 *expression* with *promoted* type."
189
190 The trick used here is from Clive D W Feather.
191*/
192
193#define INT8_C(val) (INT_LEAST8_MAX-INT_LEAST8_MAX+(val))
194#define INT16_C(val) (INT_LEAST16_MAX-INT_LEAST16_MAX+(val))
195#define INT32_C(val) (INT_LEAST32_MAX-INT_LEAST32_MAX+(val))
196/* The 'trick' doesn't work in C89 for long long because, without
197 suffix, (val) will be evaluated as int, not intmax_t */
198#define INT64_C(val) val##LL
199
200#define UINT8_C(val) (val)
201#define UINT16_C(val) (val)
202#define UINT32_C(val) (val##U)
203#define UINT64_C(val) val##ULL
204
205/* 7.18.4.2 Macros for greatest-width integer constants */
206#define INTMAX_C(val) val##LL
207#define UINTMAX_C(val) val##ULL
208
209#endif /* !defined ( __cplusplus) || defined __STDC_CONSTANT_MACROS */
