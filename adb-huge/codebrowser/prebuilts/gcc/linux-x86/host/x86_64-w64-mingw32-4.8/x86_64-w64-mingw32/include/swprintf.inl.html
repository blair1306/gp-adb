22/* __attribute__((__format__ (gnu_wprintf, 3, 0))) */ __MINGW_ATTRIB_NONNULL(3)
23int vswprintf (wchar_t *__stream, size_t __count, const wchar_t *__format, __builtin_va_list __local_argv)

28__mingw_ovr
29/* __attribute__((__format__ (gnu_wprintf, 3, 4))) */ __MINGW_ATTRIB_NONNULL(3)
30int swprintf (wchar_t *__stream, size_t __count, const wchar_t *__format, ...)
31{
32 register int __retval;
33 __builtin_va_list __local_argv;
34
35 __builtin_va_start( __local_argv, __format );
36 __retval = vswprintf( __stream, __count, __format, __local_argv );
37 __builtin_va_end( __local_argv );
38 return __retval;
39}
