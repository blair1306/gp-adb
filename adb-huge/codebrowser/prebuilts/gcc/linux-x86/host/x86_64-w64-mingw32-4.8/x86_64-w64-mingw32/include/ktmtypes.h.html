102typedef struct _TRANSACTION_NOTIFICATION {

109typedef struct _TRANSACTION_NOTIFICATION_RECOVERY_ARGUMENT {
110 GUID EnlistmentId;
111/* UOW is typedef'ed as GUID just above. Changed type of UOW
112 * to GUID from UOW for C++ compat. using ::UOW for C++ works
113 * too but we were reported some problems in corner cases and
114 * MSDN documents this structure with "GUID UOW;" anyway. */
115 GUID UOW;
116} TRANSACTION_NOTIFICATION_RECOVERY_ARGUMENT, *PTRANSACTION_NOTIFICATION_RECOVERY_ARGUMENT;
