24/* Helper functions to issue warnings and errors when needed. */
25extern unsigned long int __fdelt_chk (unsigned long int __d);
26extern unsigned long int __fdelt_warn (unsigned long int __d)
27 __warnattr ("bit outside of fd_set selected");
28#undef __FD_ELT
29#define __FD_ELT(d) \
30 __extension__ \
31 ({ unsigned long int __d = (d); \
32 (__builtin_constant_p (__d) \
33 ? (__d >= __FD_SETSIZE \
34 ? __fdelt_warn (__d) : (__d / __NFDBITS)) \
35 : __fdelt_chk (__d)); })
